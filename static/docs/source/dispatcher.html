// in worker
function buildAPI(obj) {
	_.each(obj,function(value,key) {
		if(_.isFunction(value)) {
			pmrpc.register({
				publicProcedureName: key,
				procedure: _.bind(value,obj)
			});
		}
	});
}

	},

	}
});

Ext.define('App.WorkerManager',{
	singleton: true,
	create : function(config, defaultType){
        if (config.init) {
            return config;
        } else {
            return Ext.createByAlias('worker.' + (config.ptype || defaultType), config);
        }
    },
})

Ext.define('App.LocalWorker',{
	alias: 'worker.local',
	extend: 'App.AbstractWorker',
	init: function(base) {
		this.base = base;
	},
	serialize: function() {
		return this.base.serialize();
	},
	deserialize: function(state) {
		this.base.deserialize(state);
	},
	call: function(publicProcedureName,params,cb) {
		cb(this.base[publicProcedureName].apply(this.base,params));
	},
});

////////////////////////////////////////////////////////////////////////

Ext.define('App.WebWorker',{
	alias: 'worker.web',
	extend: 'App.AbstractWorker',
	init: function(base) {
		this.worker = new Worker(this.workerUri);
	},
	call: function(publicProcedureName,params,cb) {
		pmrpc.call({
			'publicProcedureName': publicProcedureName,
			'params':params,
			'destination':this.worker,
			'onSuccess':cb,
		});
	}
});
